    P(A) = 0,4 \qquad P(B) = 0,3 \qquad \text{und} \qquad P(A\cap B) = 0,12
$$

    \item Sind die Ereignisse $A$ und $B$ stochastisch unabhängig?

    \begin{enumerate}
        \item $\overline{A}$
        \item $A \cup B$
        \item $A \cap \overline{B}$
        \item $\overline{A} \cup \overline{B}$
    \end{enumerate}
\end{enumerate}
